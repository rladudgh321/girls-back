User-agent: *
Disallow: /api
Disallow: /uploads

server {
        listen 80;
        server_name api.saranggirls.com www.saranggirls.com saranggirls.com;

        # HTTP 요청을 HTTPS로 리디렉션
        return 301 https://$host$request_uri;
    }

    server {
        listen 443 ssl;
        server_name api.saranggirls.com www.saranggirls.com saranggirls.com;

        ssl_certificate /etc/letsencrypt/live/api.saranggirls.com/fullchain.pem;
        ssl_certificate_key /etc/letsencrypt/live/api.saranggirls.com/privkey.pem;

        # CORS 설정
        add_header 'Access-Control-Allow-Origin' 'https://api.saranggirls.com https://www.saranggirls.com https://saranggirls.com' always;
        add_header 'Access-Control-Allow-Methods' 'GET, POST, PUT, DELETE, OPTIONS' always;
        add_header 'Access-Control-Allow-Headers' 'Origin, X-Requested-With, Content-Type, Accept, Authorization' always;
        add_header 'Access-Control-Allow-Credentials' 'true' always;

        # CORS Preflight 요청을 처리 (OPTIONS 요청 허용)
        if ($request_method = 'OPTIONS') {
            return 204;
        }

        location / {
            proxy_pass http://backend;  # 내부 프록시 설정
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection 'upgrade';
            proxy_set_header Host $host;
            proxy_cache_bypass $http_upgrade;
        }
    }
